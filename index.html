GPT API集成代码示例（Python）


import requests
import json

# 4Z API服务配置
API_KEY = "your_4z_api_key"  # 在控制台获取
API_URL = "https://api.zzzzapi.com/v1/chat/completions"  # GPT API中转站接口

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

# 请求参数配置
payload = {
    "model": "gpt-4o",  # 可选gpt-4o, gpt-4-turbo, gpt-3.5-turbo等
    "messages": [
        {"role": "system", "content": "你是一个专业的AI助手"},
        {"role": "user", "content": "请编写一个Python函数，实现快速排序算法"}
    ],
    "temperature": 0.7,
    "max_tokens": 1000
}

# 发送请求
try:
    response = requests.post(API_URL, headers=headers, json=payload, timeout=30)
    result = response.json()
    
    # 输出结果
    if 'choices' in result:
        content = result['choices'][0]['message']['content']
        print(content)
    else:
        print(f"API调用失败: {result}")
        
except Exception as e:
    print(f"请求异常: {str(e)}")
